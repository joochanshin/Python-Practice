def z(n, c): 
    if n == 0: 
        return Complex(0, 0) 
    else: 
        tmp = z(n - 1, c) 
        return tmp * tmp + c 
 
 
def accept_c(c): 
    if abs(z(10, c)) > 2: 
        return False 
    else:
